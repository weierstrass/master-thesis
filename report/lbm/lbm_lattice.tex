symmetry... multi speed... weights ... D2Q9

\begin{equation}\label{eq:lbm:weights}
w_i = 
\left\{
  \begin{array}{l l}
    4/9 & \quad \text{if $i = 0$}\\ 
    1/9 & \quad \text{if $i = 1,2,3,4$}\\    
    1/36 & \quad \text{if $i = 5,6,7,8$}\\
  \end{array} \right.
\end{equation}
